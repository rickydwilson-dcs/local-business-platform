/**

export function generateImageAlt(
  serviceName: string,
  locationName?: string,
  context?: string
): string {
  const base = serviceName;
  const location = locationName ? ` in ${locationName}` : " in South East UK";

export function generateImageAlt(

): string {
  const base = serviceName;

  const location = locationName ? ` in ${locationName}` : " services";
